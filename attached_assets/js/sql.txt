CREATE TABLE achievements(
    aid INT NOT NULL PRIMARY KEY AUTO_INCREMENT,
    aname VARCHAR(255) NOT NULL,
    astart INT NOT NULL
);

CREATE TABLE users (
    id INT NOT NULL PRIMARY KEY AUTO_INCREMENT,
    username VARCHAR(50) NOT NULL UNIQUE,
    password VARCHAR(255) NOT NULL,
    email VARCHAR(255) NOT NULL,
    totalplayed VARCHAR(255) NOT NULL,
    levelid INT NOT NULL,
    FOREIGN KEY (levelid) REFERENCES achievements(aid),
    created_at DATETIME DEFAULT CURRENT_TIMESTAMP
);

CREATE TABLE scores (
    id INT NOT NULL PRIMARY KEY AUTO_INCREMENT,
    elapsedTime VARCHAR(200) NOT NULL,
    scoredate VARCHAR(255) NOT NULL,
    score INT NOT NULL DEFAULT 0,
    uid INT NOT NULL,
    FOREIGN KEY (uid) REFERENCES users(id)
);

CREATE TABLE questions_answers(
    qid INT NOT NULL PRIMARY KEY AUTO_INCREMENT,
    question VARCHAR(255) NOT NULL,
    answer VARCHAR(255) NOT NULL
);

//Display all-time top scores per user
SELECT u.username,MAX(t.score) FROM users AS u INNER JOIN (SELECT s.score,s.elapsedTime,s.scoredate,s.uid FROM scores AS s) AS t ON u.id = t.uid GROUP BY u.username ORDER BY t.score DESC;

//Display all time top scores per player with details:
SELECT a.username,a.tscore,s.elapsedTime,s.scoredate FROM (SELECT u.username,MAX(t.score) AS tscore FROM users AS u INNER JOIN (SELECT s.score,s.elapsedTime,s.scoredate,s.uid FROM scores AS s) AS t ON u.id = t.uid GROUP BY u.username ORDER BY t.score DESC) AS a INNER JOIN scores AS s ON s.score = a.tscore ORDER BY a.tscore DESC LIMIT 10;
